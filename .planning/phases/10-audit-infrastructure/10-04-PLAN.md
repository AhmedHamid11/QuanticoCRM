```go
package middleware

import (
    "github.com/fastcrm/backend/internal/service"
    "github.com/gofiber/fiber/v2"
)

// AuditAuthorizationFailures creates middleware that logs 403 responses
func AuditAuthorizationFailures(auditLogger *service.AuditLogger) fiber.Handler {
    return func(c *fiber.Ctx) error {
        // Execute the next handler first
        err := c.Next()

        // Check if response was 403 Forbidden
        if c.Response().StatusCode() == fiber.StatusForbidden {
            // Extract context values (may not exist if auth failed early)
            userID, _ := c.Locals("userID").(string)
            email, _ := c.Locals("email").(string)
            orgID, _ := c.Locals("orgID").(string)

            // Log asynchronously to not block response
            go auditLogger.LogAuthorizationDenied(
                c.Context(),
                userID,
                email,
                orgID,
                c.Path(),
                c.Method(),
                c.IP(),
                c.Get("User-Agent"),
            )
        }

        return err
    }
}
```

Ensure LogAuthorizationDenied exists in audit service (from Plan 01):

- Details: {"path": ..., "method": ...}

If not present, add it to service/audit.go:
```go
func (a *AuditLogger) LogAuthorizationDenied(ctx context.Context, actorID, actorEmail, orgID, path, method, ipAddress, userAgent string) {
    a.Log(ctx, AuditEvent{
        EventType:  AuditEventAuthorizationDenied,
        ActorID:    actorID,
        ActorEmail: actorEmail,
        OrgID:      orgID,
        TargetID:   path,
        TargetType: "endpoint",
        IPAddress:  ipAddress,
        UserAgent:  userAgent,
        Success:    false,
        Details: map[string]interface{}{
            "method": method,
            "path":   path,
        },
    })
}
```
