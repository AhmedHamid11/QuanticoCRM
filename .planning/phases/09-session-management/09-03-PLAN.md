```typescript
// Session timeout tracking for Quantico CRM using Svelte 5 runes
// DECISION: User must explicitly click "Stay logged in" - activity does NOT extend session
import { PUBLIC_API_URL } from '$env/static/public';
import { auth, logoutWithSessionExpired } from './auth.svelte';

interface SessionState {
    warningVisible: boolean;
    secondsRemaining: number;
    expiryType: 'idle' | 'absolute';
    idleTimeoutMs: number;      // From org settings, default 30 * 60 * 1000
    absoluteTimeoutMs: number;  // From org settings, default 24 * 60 * 60 * 1000
    sessionStart: number;       // Timestamp when session started
    lastActivity: number;       // Timestamp of last tracked activity
    returnUrl: string | null;   // URL to return to after re-login
}

const WARNING_THRESHOLD_MS = 5 * 60 * 1000; // 5 minutes before expiry

let sessionState = $state<SessionState>({
    warningVisible: false,
    secondsRemaining: 300,
    expiryType: 'idle',
    idleTimeoutMs: 30 * 60 * 1000,
    absoluteTimeoutMs: 24 * 60 * 60 * 1000,
    sessionStart: Date.now(),
    lastActivity: Date.now(),
    returnUrl: null,
});

let checkInterval: number | null = null;

// Track user activity (clicks and keystrokes only, per CONTEXT.md)
// NOTE: This updates local tracking only - does NOT call backend or extend session
function trackActivity() {
    if (!auth.isAuthenticated) return;
    sessionState.lastActivity = Date.now();
    // If warning was showing and user interacted, they need to click the button
    // Activity alone does NOT dismiss warning or extend session
}

// Check if session is approaching timeout
function checkSessionTimeout() {
    if (!auth.isAuthenticated) {
        hideWarning();
        return;
    }

    const now = Date.now();

    // Calculate remaining time for both timeouts
    const idleElapsed = now - sessionState.lastActivity;
    const idleRemaining = sessionState.idleTimeoutMs - idleElapsed;

    const absoluteElapsed = now - sessionState.sessionStart;
    const absoluteRemaining = sessionState.absoluteTimeoutMs - absoluteElapsed;

    // Check which expires first
    const nearestExpiry = Math.min(idleRemaining, absoluteRemaining);
    const expiryType = idleRemaining <= absoluteRemaining ? 'idle' : 'absolute';

    // If expired, force logout
    if (nearestExpiry <= 0) {
        handleSessionExpired();
        return;
    }

    // If within warning threshold, show warning
    if (nearestExpiry <= WARNING_THRESHOLD_MS) {
        sessionState.warningVisible = true;
        sessionState.secondsRemaining = Math.floor(nearestExpiry / 1000);
        sessionState.expiryType = expiryType;
    } else if (sessionState.warningVisible && nearestExpiry > WARNING_THRESHOLD_MS) {
        // Warning should stay visible until user clicks button or session expires
        // Don't auto-hide based on time alone
    }
}

// Called when user explicitly clicks "Stay logged in"
async function extendSession(): Promise<boolean> {
    try {
        const API_BASE = PUBLIC_API_URL || '/api/v1';
        const response = await fetch(`${API_BASE}/auth/extend-session`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': `Bearer ${auth.accessToken}`,
            },
            credentials: 'include',
        });

        if (response.ok) {
            // Reset tracking
            sessionState.lastActivity = Date.now();
            sessionState.warningVisible = false;
            sessionState.secondsRemaining = 300;
            return true;
        }

        // If extend fails (maybe session already expired), force re-login
        if (response.status === 401) {
            handleSessionExpired();
        }
        return false;
    } catch {
        return false;
    }
}

function hideWarning() {
    sessionState.warningVisible = false;
}

function handleSessionExpired() {
    // Capture current URL for return after login
    if (typeof window !== 'undefined') {
        sessionState.returnUrl = window.location.pathname + window.location.search;
    }

    hideWarning();
    stopSessionTracking();
    logoutWithSessionExpired();
}

// Initialize session tracking when user logs in
function initSessionTracking(idleTimeoutMinutes: number = 30, absoluteTimeoutMinutes: number = 1440) {
    sessionState.idleTimeoutMs = idleTimeoutMinutes * 60 * 1000;
    sessionState.absoluteTimeoutMs = absoluteTimeoutMinutes * 60 * 1000;
    sessionState.sessionStart = Date.now();
    sessionState.lastActivity = Date.now();
    sessionState.warningVisible = false;

    // Set up activity listeners
    if (typeof window !== 'undefined') {
        window.addEventListener('click', trackActivity);
        window.addEventListener('keydown', trackActivity);
        // NOTE: No 'mousemove' per CONTEXT.md - only clicks and typing

        // Check session every 30 seconds
        if (checkInterval) clearInterval(checkInterval);
        checkInterval = window.setInterval(checkSessionTimeout, 30 * 1000);
    }
}

function stopSessionTracking() {
    if (typeof window !== 'undefined') {
        window.removeEventListener('click', trackActivity);
        window.removeEventListener('keydown', trackActivity);
        if (checkInterval) {
            clearInterval(checkInterval);
            checkInterval = null;
        }
    }
}

// Export reactive getters and actions
export const session = {
    get warningVisible() { return sessionState.warningVisible; },
    get secondsRemaining() { return sessionState.secondsRemaining; },
    get expiryType() { return sessionState.expiryType; },
    get returnUrl() { return sessionState.returnUrl; },
};

export {
    sessionState,
    initSessionTracking,
    stopSessionTracking,
    extendSession,
    checkSessionTimeout,
    handleSessionExpired,
};
```

<!-- Session timeout warning toast -->
<script lang="ts">
    import { session, sessionState, extendSession, handleSessionExpired } from '$lib/stores/session.svelte';

    let isExtending = $state(false);

    // Countdown effect - decrement every second when warning visible
    $effect(() => {
        if (!sessionState.warningVisible) return;

        const interval = setInterval(() => {
            if (sessionState.secondsRemaining > 0) {
                sessionState.secondsRemaining--;
            } else {
                clearInterval(interval);
                handleSessionExpired();
            }
        }, 1000);

        return () => clearInterval(interval);
    });

    function formatTime(seconds: number): string {
        const mins = Math.floor(seconds / 60);
        const secs = seconds % 60;
        return `${mins}:${secs.toString().padStart(2, '0')}`;
    }

    async function handleStayLoggedIn() {
        isExtending = true;
        try {
            await extendSession();
        } finally {
            isExtending = false;
        }
    }
</script>

{#if sessionState.warningVisible}

            <div class="flex-shrink-0">

                </p>
