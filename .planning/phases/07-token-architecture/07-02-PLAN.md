<objective>
Implement HttpOnly cookie-based refresh token storage on the backend.

Purpose: HttpOnly cookies cannot be accessed by JavaScript, making refresh tokens immune to XSS attacks. Even if an attacker injects malicious JavaScript, they cannot steal the refresh token to impersonate the user.

Output: Auth handlers updated to set/read refresh tokens via HttpOnly cookies instead of request/response body. Cookie helper functions for consistent cookie configuration.
</objective>

Add cookie helper functions to security.go for consistent HttpOnly cookie handling:

```go
import (
    "time"
    // ... existing imports
)

// RefreshTokenCookieName is the name of the refresh token cookie.
const RefreshTokenCookieName = "refresh_token"

// RefreshTokenExpiry is the expiry duration for refresh tokens.
// Must match the service layer's RefreshTokenExpiry.
const RefreshTokenExpiry = 7 * 24 * time.Hour // 7 days

// SetRefreshTokenCookie sets the refresh token as an HttpOnly cookie.
// The cookie is:
// - HttpOnly: Cannot be accessed by JavaScript (XSS protection)
// - Secure: Only sent over HTTPS in production
// - SameSite=Strict: Prevents CSRF attacks
// - Path=/api/v1/auth: Only sent to auth endpoints (minimizes exposure)
func SetRefreshTokenCookie(c *fiber.Ctx, token string, isProduction bool) {
    c.Cookie(&fiber.Cookie{
        Name:     RefreshTokenCookieName,
        Value:    token,
        Path:     "/api/v1/auth",
        Expires:  time.Now().Add(RefreshTokenExpiry),
        HTTPOnly: true,
        Secure:   isProduction,
        SameSite: "Strict",
    })
}

// ClearRefreshTokenCookie clears the refresh token cookie.
// Used on logout or when token reuse is detected.
func ClearRefreshTokenCookie(c *fiber.Ctx, isProduction bool) {
    c.Cookie(&fiber.Cookie{
        Name:     RefreshTokenCookieName,
        Value:    "",
        Path:     "/api/v1/auth",
        Expires:  time.Now().Add(-time.Hour), // Expire immediately
        HTTPOnly: true,
        Secure:   isProduction,
        SameSite: "Strict",
    })
}

// GetRefreshTokenFromCookie reads the refresh token from the HttpOnly cookie.
// Returns empty string if cookie is not present.
func GetRefreshTokenFromCookie(c *fiber.Ctx) string {
    return c.Cookies(RefreshTokenCookieName)
}
```

These helpers ensure consistent cookie configuration across all auth endpoints.

```go
func (h *AuthHandler) RefreshToken(c *fiber.Ctx) error {
    // Read refresh token from HttpOnly cookie (not body)
    refreshToken := middleware.GetRefreshTokenFromCookie(c)
    if refreshToken == "" {
        return c.Status(fiber.StatusUnauthorized).JSON(fiber.Map{
            "error": "Refresh token not found",
        })
    }

    response, err := h.authService.RefreshTokens(c.Context(), refreshToken)
    if err != nil {
        // On any auth error, clear the cookie
        middleware.ClearRefreshTokenCookie(c, h.isProduction)
        return h.handleAuthError(c, err)
    }

    // Set new refresh token cookie (rotation)
    middleware.SetRefreshTokenCookie(c, response.RefreshToken, h.isProduction)

    // Return new access token in body
    return c.JSON(fiber.Map{
        "accessToken": response.AccessToken,
        "expiresAt":   response.ExpiresAt,
        "user":        response.User,
    })
}
```

```go
    refreshToken := middleware.GetRefreshTokenFromCookie(c)

    if refreshToken != "" {
        if err := h.authService.Logout(c.Context(), refreshToken); err != nil {
            log.Printf("Logout error: %v", err)
            // Continue to clear cookie even if server-side logout fails
        }
    }

    // Always clear the cookie
    middleware.ClearRefreshTokenCookie(c, h.isProduction)

    return c.JSON(fiber.Map{
        "message": "Logged out successfully",
    })
}
```
