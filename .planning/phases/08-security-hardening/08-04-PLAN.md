Create middleware that blocks users with mustChangePassword=true from accessing protected routes.

1. Create middleware/password_change.go:
```go
package middleware

import (
    "strings"

    "github.com/gofiber/fiber/v2"
)

// RequirePasswordChange returns middleware that enforces password change for flagged users.
// Users with mustChangePassword=true in their JWT will receive 403 on protected routes.
// Allowed endpoints: change-password, logout, me (for auth state), health
func RequirePasswordChange() fiber.Handler {
    return func(c *fiber.Ctx) error {
        // Skip for unauthenticated routes (no user in Locals)
        mustChange, ok := c.Locals("mustChangePassword").(bool)
        if !ok || !mustChange {
            return c.Next()
        }

        // Skip for API tokens (set by auth middleware)
        if isAPIToken, _ := c.Locals("isAPIToken").(bool); isAPIToken {
            return c.Next()
        }

        // Allow specific endpoints during forced password change
        path := c.Path()
        allowedPaths := []string{
            "/api/v1/auth/change-password",
            "/api/v1/auth/logout",
            "/api/v1/auth/me",
            "/api/v1/health",
        }
        for _, allowed := range allowedPaths {
            if strings.HasPrefix(path, allowed) {
                return c.Next()
            }
        }

        // Block all other requests
        return c.Status(403).JSON(fiber.Map{
            "error":   "Password change required",
            "code":    "PASSWORD_CHANGE_REQUIRED",
            "message": "Your password doesn't meet our updated security requirements. Please change it to continue.",
        })
    }
}
```

<script lang="ts">
    import { goto } from '$app/navigation';
    import { page } from '$app/stores';
    import PasswordInput from '$lib/components/PasswordInput.svelte';
    import { auth } from '$lib/stores/auth';
    import { api } from '$lib/api';

    let currentPassword = $state('');
    let newPassword = $state('');
    let confirmPassword = $state('');
    let error = $state('');
    let loading = $state(false);

    // Check if this is a forced change (from URL param or auth state)
    const isForced = $derived($page.url.searchParams.get('required') === 'true' || $auth.mustChangePassword);

    async function handleSubmit(e: Event) {
        e.preventDefault();
        error = '';

        // Validate
        if (newPassword.length < 8) {
            error = 'New password must be at least 8 characters';
            return;
        }
        if (newPassword.length > 128) {
            error = 'New password must be 128 characters or less';
            return;
        }
        if (newPassword !== confirmPassword) {
            error = 'New passwords do not match';
            return;
        }

        loading = true;
        try {
            await api.post('/auth/change-password', {
                currentPassword,
                newPassword
            });

            // Refresh auth state to get new token without mustChangePassword
            await auth.refresh();

            // Redirect to dashboard
            goto('/');
        } catch (err: any) {
            error = err.message || 'Failed to change password';
        } finally {
            loading = false;
        }
    }
</script>
