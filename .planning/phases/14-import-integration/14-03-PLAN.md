```go
            switch resolution.Action {
            case "skip":

                auditEntries = append(auditEntries, AuditEntry{

                    Reason: "User chose to skip this duplicate",

            case "update":
                recordsToUpdate = append(recordsToUpdate, struct {

                    TargetID string
                }{Record: record, TargetID: resolution.SelectedMatchID})
                auditEntries = append(auditEntries, AuditEntry{

                    Reason: "User chose to update existing record",
                })
                continue
            case "merge":
                mergedCount++
                auditEntries = append(auditEntries, AuditEntry{
                    RowIndex: i, Action: "merge", MatchedID: resolution.SelectedMatchID,
                    Reason: "User sent to merge wizard",
                })
                continue
            case "import":
                auditEntries = append(auditEntries, AuditEntry{
                    RowIndex: i, Action: "import", Reason: "User chose to import anyway",
                })
                // Fall through to create
            }
        }
    }

    // Check within-file skip indices
    if contains(options.WithinFileSkipIndices, i) {
        skippedCount++
        auditEntries = append(auditEntries, AuditEntry{
            RowIndex: i, Action: "skip",
            Reason: "Within-file duplicate (another row from same group was kept)",
        })
        continue
    }
```

```go
// AuditEntry represents one row's resolution in the audit report
type AuditEntry struct {
    RowIndex  int    `json:"rowIndex"`
    Action    string `json:"action"`
    MatchedID string `json:"matchedId,omitempty"`
    Reason    string `json:"reason"`
}

// GenerateAuditReport creates a CSV-formatted audit report of import resolutions
func (s *ImportDuplicateService) GenerateAuditReport(entries []AuditEntry, result interface{}) []byte {
    var buf bytes.Buffer
    writer := csv.NewWriter(&buf)

    // Header
    writer.Write([]string{"Row Number", "Action", "Matched Record ID", "Reason"})

    // Entries
    for _, entry := range entries {
        writer.Write([]string{
            fmt.Sprintf("%d", entry.RowIndex+1),
            entry.Action,
            entry.MatchedID,
            entry.Reason,
        })
    }

    // Summary section
    writer.Write([]string{})
    writer.Write([]string{"Summary"})
    // Counts will be populated based on the import response

    writer.Flush()
    return buf.Bytes()
}
```

Add base64 helper if not already present:

```typescript
function downloadAuditReport(base64Data: string) {
    const binaryString = atob(base64Data);
    const bytes = new Uint8Array(binaryString.length);
    for (let i = 0; i < binaryString.length; i++) {
        bytes[i] = binaryString.charCodeAt(i);
    }
    const blob = new Blob([bytes], { type: 'text/csv' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `import-audit-${entityName}-${new Date().toISOString().split('T')[0]}.csv`;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
}
```
